program ::= expr_lst
expr_lst ::= exprs | exprs expr_lst
exprs ::= '(' expr ')' | value
expr ::= value | operation | bind | exprs
operation ::= op expr expr
bind ::= 'def' id exprs
if_expr ::= 'if' exprs exprs exprs
fn_call ::= id exprs
op ::= '+' | '-' | '*' | '/'
value ::= [0-9]+
id ::= [a-zA-Z]+
